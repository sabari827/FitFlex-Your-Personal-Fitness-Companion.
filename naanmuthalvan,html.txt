    <!-- JavaScript for BMI Calculator -->
    <script>
        document.getElementById('calculate-bmi').addEventListener('click', function() {
            const weight = parseFloat(document.getElementById('weight').value);
            const height = parseFloat(document.getElementById('height').value);
            if (weight && height && height > 0) {
                const bmi = (weight / (height * height)).toFixed(2);
                let interpretation = '';
                if (bmi < 18.5) interpretation = 'Underweight - Consider gaining weight healthily.';
                else if (bmi < 24.9) interpretation = 'Normal weight - Great job!';
                else if (bmi < 29.9) interpretation = 'Overweight - Focus on healthy habits.';
                else interpretation = 'Obese - Consult a professional for guidance.';
                document.getElementById('bmi-result').innerHTML = <p>BMI: ${bmi}</p><p>${interpretation}</p>;
            } else {
                document.getElementById('bmi-result').innerHTML = '<p>Please enter valid values.</p>';
            }
        });
    </script>
